import random

app = Flask(__name__)

# 1. ХИСОБ БО МУХОФИЗАТ
def calc_rsi(prices, period=14):
    try:
        if len(prices) < period + 2: return 50.0
        changes = [prices[i] - prices[i-1] for i in range(1, len(prices))]
        gains = [c for c in changes[-period:] if c > 0]
        losses = [-c for c in changes[-period:] if c < 0]
        avg_gain = sum(gains) / period if gains else 0.0
        avg_loss = sum(losses) / period if losses else 0.0
        if avg_loss == 0.0: return 100.0 if avg_gain > 0 else 50.0
        rs = avg_gain / avg_loss
        return 100.0 - (100.0 / (1.0 + rs))
    except: return 50.0

def calc_ema(prices, period):
    try:
        if len(prices) < period: return prices[-1]
        k = 2.0 / (period + 1.0)
        ema = sum(prices[:period]) / float(period)
        for p in prices[period:]: ema = p * k + ema * (1.0 - k)
        return ema
    except: return prices[-1]

# 2. СИСТЕМАИ 8 ИНДИКАТОР
def get_signal(coin):
    base = {'BTC': 67000, 'ETH': 3500, 'LTC': 90, 'TRX': 0.15, 'SOL': 150}.get(coin, 100)
    prices = [base + random.uniform(-base*0.01, base*0.01) for _ in range(200)]
    price = prices[-1]
    rsi = calc_rsi(prices, 14)
    ema20 = calc_ema(prices, 20)
    ema50 = calc_ema(prices, 50)
    vwap = sum(prices[-20:]) / 20.0
    macd = calc_ema(prices, 12) - calc_ema(prices, 26)
    atr = max(prices[-14:]) - min(prices[-14:]) if len(prices) > 14 else 1.0

    score = 0
    if rsi < 30: score = score + 2
    if rsi > 70: score = score - 2
    if macd > 0: score = score + 2
    if ema20 > ema50: score = score + 2
    if price > vwap: score = score + 1
    if price > ema50: score = score + 1

    if score >= 5: s = "ПОКУПАТЬ СИЛЬНО"; c = "#00ff88"
    elif score >= 2: s = "ПОКУПАТЬ"; c = "#66ff66"
    elif score <= -5: s = "ПРОДАВАТЬ СИЛЬНО"; c = "#ff4444"
    elif score <= -2: s = "ПРОДАВАТЬ"; c = "#ff6666"
    else: s = "ЖДАТЬ"; c = "#ffaa00"
    return price, rsi, ema20, ema50, macd, atr, s, c, score

@app.route('/')
def home():
    try:
        coin = request.args.get('coin', 'BTC')
        p, r, e20, e50, m, a, s, c, sc = get_signal(coin)
        trend = "BULL" if e20 > e50 else "BEAR"
